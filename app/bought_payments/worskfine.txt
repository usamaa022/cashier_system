  // FIX 1: Improved compressImage - converts to grayscale and compresses reliably
  const compressImage = async (file) => {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.readAsDataURL(file);
      reader.onload = (event) => {
        const img = new Image();
        img.src = event.target.result;
        img.onload = () => {
          const canvas = document.createElement("canvas");
          let width = img.width;
          let height = img.height;

          const maxDim = 1000;
          if (width > height && width > maxDim) {
            height = Math.round((height * maxDim) / width);
            width = maxDim;
          } else if (height > width && height > maxDim) {
            width = Math.round((width * maxDim) / height);
            height = maxDim;
          } else if (width > maxDim) {
            height = Math.round((height * maxDim) / width);
            width = maxDim;
          }

          canvas.width = width;
          canvas.height = height;
          const ctx = canvas.getContext("2d");
          ctx.drawImage(img, 0, 0, width, height);

          // Convert to grayscale
          const imageData = ctx.getImageData(0, 0, width, height);
          const data = imageData.data;
          for (let i = 0; i < data.length; i += 4) {
            const gray = Math.round(0.299 * data[i] + 0.587 * data[i + 1] + 0.114 * data[i + 2]);
            data[i] = gray;
            data[i + 1] = gray;
            data[i + 2] = gray;
          }
          ctx.putImageData(imageData, 0, 0);

          // Try quality 0.7 first
          canvas.toBlob(
            (blob) => {
              if (!blob) {
                reject(new Error("Failed to process image"));
                return;
              }
              if (blob.size <= 1024 * 1024) {
                resolve(
                  new File(
                    [blob],
                    file.name.replace(/\.[^/.]+$/, "") + "_gray.jpg",
                    { type: "image/jpeg" }
                  )
                );
              } else {
                // Try lower quality 0.4
                canvas.toBlob(
                  (blob2) => {
                    if (!blob2) {
                      reject(new Error("Failed to compress image"));
                      return;
                    }
                    resolve(
                      new File(
                        [blob2],
                        file.name.replace(/\.[^/.]+$/, "") + "_gray.jpg",
                        { type: "image/jpeg" }
                      )
                    );
                  },
                  "image/jpeg",
                  0.4
                );
              }
            },
            "image/jpeg",
            0.7
          );
        };
        img.onerror = () => reject(new Error("Failed to load image"));
      };
      reader.onerror = () => reject(new Error("Failed to read image file"));
    });
  };

          if (!uniqueReturnsMap.has(returnItem.id)) {
            uniqueReturnsMap.set(returnItem.id, {

              returnItem.name &&
              !existing.items.some((i) => i.barcode === returnItem.barcode)

                name: returnItem.name,
                barcode: returnItem.barcode,

                returnPriceIQD: returnItem.returnPriceIQD,

              existing.totalReturn +=
                (returnItem.returnPrice || 0) * (returnItem.returnQuantity || 0);

        });

        setReturns(uniqueReturns);

        console.error("Error loading company data:", error);
        setError("Failed to load company data");
      } finally {
        setLoading(false);
      }
    };
    loadCompanyData();
  }, [selectedCompany, isEditMode, initialLoadComplete]);

        } else {
          boughtIQD += bill.totalAmountIQD || bill.totalAmount || 0;
        }

              returnIQD += itemTotal;

          } else {

      boughtIQD,
      returnUSD,

      netIQD: boughtIQD - returnIQD,

        setTimeout(() => hardcopyBillNumberRef.current?.focus(), 400);
      }
      hasError = true;
    }

    if (selectedBoughtBills.length === 0 && selectedBoughtReturns.length === 0) {
      if (!hasError) setError("Please select at least one bill or return to process");
      hasError = true;
    }

    if (hasError) return;

        setImageUploading(true);
        try {
          let processedImage = billImage;

          // Always compress/grayscale regardless of size for consistency

          if (processedImage.size > 1024 * 1024) {
            throw new Error("Image is still over 1MB after compression. Please use a smaller image.");
          }

          const storage = getStorage();
          const fileName = `boughtPayments/${Date.now()}_${processedImage.name.replace(
            /[^a-zA-Z0-9._]/g,
            "_"
          )}`;
          const storageRef = ref(storage, fileName);

          const metadata = {
            contentType: "image/jpeg",
            customMetadata: {
              originalName: billImage.name,
              uploadDate: new Date().toISOString(),
              companyId: selectedCompany,
              compressed: "true",
              greyscale: "true",
            },
          };

          const uploadResult = await uploadBytes(storageRef, processedImage, metadata);
          imageUrl = await getDownloadURL(uploadResult.ref);
        } catch (uploadError) {
          console.error("Image upload error:", uploadError);
          setError(uploadError.message || "Failed to upload image. Please try again or skip the image.");
          setSubmitting(false);
          setImageUploading(false);
          return;
        } finally {
          setImageUploading(false);
        }
      }

      const paymentNumber = `BPAY-${Date.now()}-${Math.floor(Math.random() * 1000)}`;

        returnTotalUSD: currencyTotals.returnUSD,
        returnTotalIQD: currencyTotals.returnIQD,

    setShowPaymentModal(false);
    setSelectedPayment(null);

    setSelectedImageUrl(imageUrl);
    setShowImageModal(true);
  };

    setShowImageModal(false);
    setSelectedImageUrl("");

    try {
      const allBoughtBills = await getBoughtBills();
      const allReturns = await getReturnsForCompany(payment.companyId);
      const boughtDetails = allBoughtBills.filter((bill) =>
        payment.selectedBoughtBills?.includes(bill.id)

        payment.selectedBoughtReturns?.includes(r.id)

      printWindow.document.close();

      console.error("Error generating print:", error);
      setError("Failed to generate print preview");
    } finally {
      setPrintLoading(false);
    }

  // FIX 4: Logo added to print header, improved layout
  const generatePrintHTML = (payment, boughtBillsList, returnsList) => {
    let totalBoughtUSD = 0;
    let totalBoughtIQD = 0;
    let totalReturnUSD = 0;
    let totalReturnIQD = 0;

    const boughtItemsTable = [];
    const returnItemsTable = [];

    if (boughtBillsList && boughtBillsList.length > 0) {
      boughtBillsList.forEach((bill) => {
        const currency = bill.currency || "USD";
        let billTotal = 0;

        if (bill.items && Array.isArray(bill.items) && bill.items.length > 0) {
          bill.items.forEach((item) => {
            billTotal += (item.basePrice || item.price || 0) * (item.quantity || 1);
          });
        } else if (bill.totalAmountUSD || bill.totalAmountIQD) {
          billTotal =
            currency === "USD" ? bill.totalAmountUSD || 0 : bill.totalAmountIQD || 0;
        } else if (bill.totalAmount) {
          billTotal = bill.totalAmount;
        } else if (bill.amount) {
          billTotal = bill.amount;
        }

        if (currency === "USD") totalBoughtUSD += billTotal;
        else totalBoughtIQD += billTotal;

        const displayAmount =
          currency === "USD" ? formatUSD(billTotal) : formatCurrency(billTotal);

        boughtItemsTable.push(`
          <tr>
            <td style="padding:6px 8px;border-bottom:1px solid #e5e7eb;">${bill.billNumber || bill.id}</td>
            <td style="padding:6px 8px;border-bottom:1px solid #e5e7eb;">${formatDateToDMY(bill.date)}</td>
            <td style="padding:6px 8px;border-bottom:1px solid #e5e7eb;text-align:right;font-weight:bold;color:${currency === "USD" ? "#059669" : "#2563eb"};">
              +${displayAmount}
            </td>
          </tr>
        `);

                  </thead>
                  <tbody>
                    {paymentDetails[selectedPayment.id].returns.map((ret) => {
                      const currency = ret.displayCurrency || ret.currency || "USD";
                      const amount =
                        ret.displayAmount ||
                        (currency === "USD" ? ret.calculatedAmountUSD : ret.calculatedAmountIQD) ||
                        0;
                      const hasUSD = ret.calculatedAmountUSD > 0;
                      const hasIQD = ret.calculatedAmountIQD > 0;

                      return (
                        <tr key={ret.id}>
                          <td style={{ padding: "8px", fontSize: "0.8rem" }}>
                            {ret.returnBillNumber || ret.returnNumber || ret.id?.slice(-6)}
                          </td>
                          <td style={{ padding: "8px", fontSize: "0.8rem" }}>
                            {formatDateToDMY(ret.returnDate || ret.date)}
                          </td>
                          <td style={{ padding: "8px", textAlign: "right", fontWeight: "bold", fontSize: "0.8rem" }}>
                            {hasUSD && hasIQD ? (
                              <div>
                                <div style={{ color: "#dc2626" }}>-{formatUSD(ret.calculatedAmountUSD)}</div>
                                <div style={{ color: "#b91c1c", fontSize: "0.7rem", marginTop: "2px" }}>
                                  -{formatCurrency(ret.calculatedAmountIQD)}
                                </div>
                              </div>
                            ) : (
                              <span style={{ color: currency === "USD" ? "#dc2626" : "#b91c1c" }}>
                                -{currency === "USD" ? formatUSD(amount) : formatCurrency(amount)}
                              </span>
                            )}
                          </td>
                        </tr>
                      );
                    })}
                  </tbody>
                </table>
              </div>
            )}

            {/* Net Amount */}
            <div
              style={{
                padding: "1rem",
                background: "#E9D5FF",
                borderRadius: "0.75rem",
                marginBottom: "1.5rem",
              }}
            >
              <div style={{ fontWeight: "bold", marginBottom: "0.5rem" }}>💰 Net Amount Paid:</div>
              <div style={{ display: "flex", gap: "1rem", flexWrap: "wrap", alignItems: "center" }}>
                {selectedPayment.netAmountUSD !== undefined &&
                  selectedPayment.netAmountUSD !== 0 && (
                    <div
                      style={{
                        fontSize: "1.25rem",
                        fontWeight: "bold",
                        color: selectedPayment.netAmountUSD > 0 ? "#059669" : "#dc2626",
                        backgroundColor:
                          selectedPayment.netAmountUSD > 0 ? "#D1FAE5" : "#FEE2E2",
                        padding: "0.5rem 1rem",
                        borderRadius: "0.5rem",
                        display: "inline-block",
                      }}
                    >
                      {selectedPayment.netAmountUSD > 0
                        ? `+${formatUSD(selectedPayment.netAmountUSD)}`
                        : formatUSD(selectedPayment.netAmountUSD)}
                    </div>
                  )}
                {selectedPayment.netAmountIQD !== undefined &&
                  selectedPayment.netAmountIQD !== 0 && (
                    <div
                      style={{
                        fontSize: "1.25rem",
                        fontWeight: "bold",
                        color: selectedPayment.netAmountIQD > 0 ? "#2563eb" : "#b91c1c",
                        backgroundColor:
                          selectedPayment.netAmountIQD > 0 ? "#DBEAFE" : "#FEE2E2",
                        padding: "0.5rem 1rem",
                        borderRadius: "0.5rem",
                        display: "inline-block",
                      }}
                    >
                      {selectedPayment.netAmountIQD > 0
                        ? `+${formatCurrency(selectedPayment.netAmountIQD)}`
                        : formatCurrency(selectedPayment.netAmountIQD)}
                    </div>
                  )}
                {(!selectedPayment.netAmountUSD || selectedPayment.netAmountUSD === 0) &&
                  (!selectedPayment.netAmountIQD || selectedPayment.netAmountIQD === 0) && (
                    <div style={{ fontSize: "1.25rem", fontWeight: "bold", color: "#6B7280" }}>
                      $0.00 / 0 IQD
                    </div>
                  )}
              </div>
            </div>

            {/* Notes */}
            {selectedPayment.notes && (
              <div
                style={{
                  marginBottom: "1.5rem",
                  padding: "1rem",
                  background: "#FFFBEB",
                  borderRadius: "0.75rem",
                }}
              >
                <strong>Notes:</strong>
                <p>{selectedPayment.notes}</p>
              </div>
            )}

            {/* Image */}
            {selectedPayment.billImageUrl && (
              <div style={{ textAlign: "center" }}>
                <img
                  src={selectedPayment.billImageUrl}
                  alt="Bill"
                  style={{
                    maxWidth: "300px",
                    maxHeight: "300px",
                    borderRadius: "0.5rem",
                    cursor: "pointer",
                  }}
                  onClick={() => handleViewImage(selectedPayment.billImageUrl)}
                />
              </div>
            )}
          </div>
        </div>
      )}

      {/* Image Modal */}
      {showImageModal && selectedImageUrl && (
        <div
          style={{
            position: "fixed",
            top: 0,
            left: 0,
            right: 0,
            bottom: 0,
            background: "rgba(0,0,0,0.8)",
            display: "flex",
            alignItems: "center",
            justifyContent: "center",
            zIndex: 2000,
          }}
          onClick={closeImageModal}
        >
          <div
            style={{
              maxWidth: "90vw",
              maxHeight: "90vh",
              background: "white",
              borderRadius: "0.5rem",
              padding: "0.5rem",
            }}
            onClick={(e) => e.stopPropagation()}
          >
            <img
              src={selectedImageUrl}
              alt="Full Bill"
              style={{ maxWidth: "100%", maxHeight: "80vh", objectFit: "contain" }}
            />
            <div style={{ textAlign: "center", marginTop: "0.5rem" }}>
              <button
                onClick={closeImageModal}
                style={{
                  padding: "0.5rem 1rem",
                  backgroundColor: colorScheme.primary,
                  color: "white",
                  border: "none",
                  borderRadius: "0.5rem",
                  cursor: "pointer",
                }}
              >
                Close
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
